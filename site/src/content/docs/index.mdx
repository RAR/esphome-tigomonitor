import { Card, CardGrid } from '@astrojs/starlight/components';

<CardGrid stagger>
  <Card title="Per-panel telemetry" icon="approve-check">
    Reads Tigo optimizer frames off the RS485 bus and publishes voltage, current,
    power, and temperature for every panel.
  </Card>
  <Card title="Built-in web UI" icon="laptop">
    A five-page web server with a JSON API — overview, per-string layout, node
    table, live status, and a YAML view. See the [Web Server &amp; API guide](/esphome-tigomonitor/guides/web-server/).
  </Card>
  <Card title="Home Assistant native" icon="puzzle">
    Sensors surface directly in Home Assistant over the ESPHome API. See the
    [Home Assistant guide](/esphome-tigomonitor/guides/home-assistant/).
  </Card>
  <Card title="Config builder" icon="setting">
    Generate a ready-to-flash ESPHome YAML for your board in the browser —
    [open the builder](/esphome-tigomonitor/config-builder/).
  </Card>
</CardGrid>
